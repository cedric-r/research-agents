    -- PART A: Create src/Output/Formatter.php --

    Namespace: App\Output. Class: Formatter.

    This class provides ANSI color constants and helper methods. It is used by repl.php for colored output AND by research.php for shared formatting (D-01).

    ANSI constants (standard codes -- NO \x01/\x02 wrapping for these, as they are used in output strings, not readline prompts):

    ```php
    // Foreground colors
    const GREEN  = "\e[32m";
    const CYAN   = "\e[36m";
    const YELLOW = "\e[33m";
    const RED    = "\e[31m";
    const WHITE  = "\e[37m";
    const RESET  = "\e[0m";

    // Text styles
    const BOLD   = "\e[1m";
    const DIM    = "\e[2m";
    ```

    Readline-safe variants (with \x01/\x02 wrapping for use in readline() prompts):
    ```php
    const RL_CYAN   = "\x01\e[36m\x02";
    const RL_BOLD   = "\x01\e[1m\x02";
    const RL_RESET  = "\x01\e[0m\x02";
    ```

    Public static helper methods (each returns a string with color codes wrapped around the input):

    - section(string $title): string -- BOLD . CYAN . $title . RESET (Per UI-SPEC 3.3: section headers in Bold Cyan)
    - winner(string $name): string -- BOLD . GREEN . $name . RESET (Per UI-SPEC 3.3: Winner in Bold Green)
    - score(float $value): string -- Colored by threshold: >=8 GREEN, >=5 YELLOW, <5 RED (Per UI-SPEC 3.3). Format: number_format($value, 1).
    - error(string $msg): string -- DIM . RED . $msg . RESET (Per UI-SPEC 3.3: errors in Dim Red)
    - agentName(string $name): string -- Cycled color based on crc32 hash: index = abs(crc32($name)) % 3, colors = [GREEN, CYAN, YELLOW] (Per UI-SPEC 3.3)
    - command(string $cmd): string -- BOLD . YELLOW . $cmd . RESET (UI-SPEC 3.5: commands in Bold Yellow)
    - separator(): string -- DIM . str_repeat('-', 72) . RESET (UI-SPEC 3.3: section separators in Dim)
    - prompt(): string -- RL_CYAN . 'research> ' . RL_RESET (UI-SPEC 3.1: prompt in bold cyan, readline-safe)
    - sessionAge(string $dateIso): string -- Returns colored string for session list. Parse ISO date. If <1h old: GREEN, <1d: YELLOW, older: DIM. (UI-SPEC 3.3: session list by recency)
    - statusLine(string $agent, string $event): string -- Formats a progress status line with agent name colored via agentName() and event in reset text. Format: "  [" . agentName($agent) . "] " . $event . RESET

    NOTE: score() expects a value on a 0-10 scale (quality scores from Arbitrator), NOT the weighted total (0-1 scale). The score formatting in research.php currently uses the composite quality score (0-10) for color thresholding. Ensure score() uses the correct scale.

    ### Main Loop:
    ```php
    $lastResult = null; // Store most recent result for /last command
    $lastDurationMs = 0;
    $lastCorrelationId = '';

    while (true) {
        $input = readline(Formatter::prompt());

        if ($input === false) { // CTRL+D = EOF
            echo PHP_EOL;
            break;
        }

        $input = trim($input);

        if ($input === '') {
            continue;
        }

        readline_add_history($input);

        if (str_starts_with($input, '/')) {
            $parts = explode(' ', $input, 2);
            $command = $parts[0];
            $arg = $parts[1] ?? '';

            match ($command) {
                '/exit', '/quit' => break 2,
                '/help'          => showHelp(),
                '/clear'         => showClear(),
                '/agents'        => showAgents($agentManager, $agentConfigs),
                '/config'        => showConfig($configLoader),
                '/history'       => showHistory($historyFile),
                '/last'          => showLast($lastResult, $lastDurationMs, $lastCorrelationId),
                '/replay'        => replaySession($arg, __DIR__),
                default          => echo "Unknown command: " . $command . PHP_EOL
                    . "Type /help for available commands." . PHP_EOL,
            };
        } else {
            $result = runResearch($input, $agentManager, $configLoader);
            if ($result !== null) {
                $lastResult = $result;
                $lastDurationMs = $result['duration_ms'] ?? 0;
                $lastCorrelationId = $result['correlation_id'] ?? '';
            }
        }
    }

    // Save history on exit (truncate to 1000 entries)
    readline_write_history($historyFile);
    // Truncate to last 1000 entries
    $hist = readline_list_history();
    if (count($hist) > 1000) {
        file_put_contents($historyFile, implode("\n", array_slice($hist, -1000)));
    }
    ```

    ### runResearch(string $question, AgentManager $agentManager, Loader $configLoader): ?array

    (Per UI-SPEC 3.7 lifecycle: Researching -> Complete -> Full output)

    ```php
    function runResearch(string $question, AgentManager $agentManager, Loader $configLoader, string $progressLogDir): ?array
    {
        $correlationId = Logger::generateCorrelationId();
        $logger = new Logger(__DIR__ . '/logs/research.log', 'REPL', $correlationId);

        try {
            // Create Arbitrator
            $arbitrator = new Arbitrator($agentManager, $configLoader, $logger);

            // Set progress log file for real-time progress (D-09)
            $slug = SessionManager::slugFromQuestion($question);
            $sessionDir = $progressLogDir . '/' . date('Y-m-d') . '_' . $slug;
            $progressLogPath = $sessionDir . '/session.log';
            $arbitrator->setProgressLogFile($progressLogPath);

            $startTime = microtime(true);

            // Show initial spinner + log in a polling loop during research
            // Because Arbitrator::research() is synchronous, we need to
            // run progress display in a parallel polling thread.
            // Approach: fork a child to run research, parent polls session.log.

            $researchPid = pcntl_fork();

            if ($researchPid === -1) {
                // Fork failed -- run synchronously (no progress display)
                $logger->warn('REPL: fork failed, running research synchronously');
                $results = $arbitrator->research($question, $correlationId);
                $durationMs = (int) ((microtime(true) - $startTime) * 1000);
                $debateResult = $arbitrator->getDebateResult();
                renderDebateOutput($results, $debateResult, $durationMs, $correlationId);
                saveSession($question, $results, $debateResult, $durationMs, $progressLogDir, $logger);
                return [
                    'results' => $results,
                    'debate' => $debateResult,
                    'duration_ms' => $durationMs,
                    'correlation_id' => $correlationId,
                ];
            }

            if ($researchPid === 0) {
                // CHILD: run research
                pcntl_signal(SIGALRM, SIG_DFL);
                pcntl_alarm(0);
                pcntl_async_signals(true);

                $childTimedOut = false;
                pcntl_signal(SIGTERM, function (int $signo) use (&$childTimedOut): void {
                    $childTimedOut = true;
                });

                $childArbitrator = new Arbitrator($agentManager, $configLoader, $logger);
                $childArbitrator->setProgressLogFile($progressLogPath);
                $childResults = $childArbitrator->research($question, $correlationId);

                // Write result to temp file
                $tempFile = sys_get_temp_dir() . '/repl_result_' . $correlationId . '.json';
                $resultData = [
                    'results' => $childResults,
                    'debate' => $childArbitrator->getDebateResult(),
                    'duration_ms' => (int) ((microtime(true) - $startTime) * 1000),
                ];
                file_put_contents($tempFile, json_encode($resultData, JSON_UNESCAPED_SLASHES), LOCK_EX);
                exit(0);
            }

            // PARENT: show progress while child runs
            $spinnerFrames = ['|', '/', '-', '\\'];
            $spinnerIndex = 0;
            $progressLogger = new ProgressLogger($progressLogPath);
            $offset = 0;
            $childRunning = true;

            while ($childRunning) {
                // Check if child finished
                $reaped = pcntl_waitpid($researchPid, $status, WNOHANG);
                if ($reaped > 0 || $reaped === -1) {
                    $childRunning = false;
                    break;
                }

                // Display spinner
                echo "\r" . $spinnerFrames[$spinnerIndex % 4] . ' Researching... (' . count(scandir(dirname($progressLogPath) !== false ? dirname($progressLogPath) : '/tmp') ?? []) . ')';
                $spinnerIndex++;

                // Read new progress lines from session.log
                $readResult = $progressLogger->readNewLines($offset);
                foreach ($readResult['lines'] as $line) {
                    $event = json_decode($line, true);
                    if ($event && isset($event['agent'], $event['event'])) {
                        $agent = $event['agent'];
                        $eventName = $event['event'];
                        echo "\n  [" . Formatter::agentName($agent !== '' ? $agent : 'system') . '] ' . $eventName;
                        if (!empty($event['data'])) {
                            $dataInfo = $event['data'];
                            if (isset($dataInfo['model'])) {
                                echo ' (' . $dataInfo['model'] . ')';
                            } elseif (isset($dataInfo['tool'])) {
                                echo ' (' . $dataInfo['tool'] . ')';
                            }
                        }
                    }
                }
                $offset = $readResult['new_offset'];

                usleep(200000); // 200ms for smooth animation
            }

            // Clear spinner line
            echo "\r\e[K";

            // Read research result from temp file
            $tempFile = sys_get_temp_dir() . '/repl_result_' . $correlationId . '.json';
            $resultData = null;
            if (file_exists($tempFile)) {
                $raw = file_get_contents($tempFile);
                if ($raw !== false) {
                    $resultData = json_decode($raw, true);
                }
                @unlink($tempFile);
            }

            if ($resultData === null) {
                echo Formatter::error('Error: Research process failed to return results.') . PHP_EOL;
                return null;
            }

            // Render full output
            $durationMs = $resultData['duration_ms'];
            $results = $resultData['results'];
            $debateResult = $resultData['debate'];
            renderDebateOutput($results, $debateResult, $durationMs, $correlationId);

            // Save session
            saveSession($question, $results, $debateResult, $durationMs, $progressLogDir, $logger);

            return [
                'results' => $results,
                'debate' => $debateResult,
                'duration_ms' => $durationMs,
                'correlation_id' => $correlationId,
            ];

        } catch (\Throwable $e) {
            echo Formatter::error('Error: ' . $e->getMessage()) . PHP_EOL;
            $logger->error('REPL research failed', ['error' => $e->getMessage()]);
            return null;
        }
    }
    ```

    ### renderDebateOutput(array $results, ?array $debateResult, int $durationMs, string $correlationId): void

    Output structure:
    1. Section separator (Formatter::separator())
    2. "SCORE TABLE" header (Formatter::section())
    3. Section separator
    4. Column headers: "Agent", "Quality", "Critique", "Diversity", "Total" (bold cyan)

    6. For each agent in debateResult['score_table']:

    8. Section separator
    9. "WINNER: {name}" in bold green
    10. Winner's full answer text (reset, no color)
    11. Model info line (dim)
    12. Section separator
    13. "JUDGE'S REASONING" header (bold cyan)
    14. Judge narrative text (reset)

    16. "== Research Complete ==" line with agents count and correlation ID

    function showHistory(string $historyFile): void
    {
        if (!file_exists($historyFile)) {
            echo 'No history yet.' . PHP_EOL;
            return;
        }

        $history = file($historyFile, FILE_IGNORE_NEW_LINES | FILE_SKIP_EMPTY_LINES);
        $history = array_filter($history, fn($line) => !str_starts_with($line, '/'));

        if (empty($history)) {
            echo 'No research questions in history.' . PHP_EOL;
            return;
        }

        // Show last 20 questions
        $recent = array_slice(array_values($history), -20);
        foreach ($recent as $i => $q) {
            echo ($i + 1) . '. ' . $q . PHP_EOL;
        }
    }

    -- PART B: Implement replaySession() in repl.php --

    ```php
    function replaySession(string $slug, string $projectRoot): void
    {
        if ($slug === '') {
            echo 'Usage: /replay <slug>' . PHP_EOL;
            echo 'Example: /replay 2026-06-13_what-is-quantum-computing' . PHP_EOL;
            return;
        }

        // Prevent path traversal (T-05-03)
        if (str_contains($slug, '/') || str_contains($slug, '..')) {
            echo Formatter::error('Invalid session slug.') . PHP_EOL;
            return;
        }

        $sessionsDir = $projectRoot . '/sessions';
        $manager = new SessionManager($sessionsDir);
        $session = $manager->readSession($slug);

        if ($session === null) {
            echo Formatter::error('Session not found: ' . $slug) . PHP_EOL;
            echo 'The session "' . $slug . '" does not exist.' . PHP_EOL;
            return;
        }

        // Display transcript with ANSI formatting
        $markdownPath = $sessionsDir . '/' . basename($slug) . '/session.md';
        $content = file_get_contents($markdownPath);
        if ($content === false) {
            echo Formatter::error('Cannot read session file.') . PHP_EOL;
            return;
        }

        // Render the markdown with ANSI:
        // Per UI-SPEC 3.5: frontmatter ignored, ## headers in Bold Cyan,
        // **Winner:** in Bold Green, errors in Dim Red.
        $lines = explode("\n", $content);
        $inFrontmatter = false;
        foreach ($lines as $line) {
            if (str_starts_with($line, '---')) {
                if (!$inFrontmatter) {
                    $inFrontmatter = true;
                    continue;
                }
                $inFrontmatter = false;
                continue;
            }

            if ($inFrontmatter) {
                continue; // Skip frontmatter
            }

            // Section headers (## ...)
            if (str_starts_with($line, '## ')) {
                echo Formatter::section($line) . PHP_EOL;
                continue;
            }

            // Winner line (**Winner:**)
            if (str_starts_with($line, '**Winner:**') || str_starts_with($line, '**WINNER:**')) {
                echo Formatter::winner($line) . PHP_EOL;
                continue;
            }

            // Error lines (lines starting with - **...** containing error)
            if (preg_match('/^- \*\*(.+?)\*\*:\s*(Error|Failed|Timeout)/i', $line)) {
                echo Formatter::error($line) . PHP_EOL;
                continue;
            }

            // Table rows (score table)
            if (str_starts_with($line, '|') && substr_count($line, '|') >= 4) {
                echo Formatter::DIM . $line . Formatter::RESET . PHP_EOL;
                continue;
            }

            echo $line . PHP_EOL;
        }
    }
    ```

    (Per UI-SPEC 3.5: Reads session.md, renders with ANSI. Frontmatter ignored, ## headers in Bold Cyan, **Winner:** in Bold Green, errors in Dim Red.)

    function saveSession(string $question, array $results, ?array $debateResult, int $durationMs, string $sessionsDir, Logger $logger): void
    {
        try {
            $manager = new SessionManager($sessionsDir);
            $data = [
                'results' => $results,
                'debate' => $debateResult,

    }
