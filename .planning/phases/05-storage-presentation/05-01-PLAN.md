        private function rmdirRecursive(string $dir): void
        {
            if (!is_dir($dir)) { return; }
            $items = new \RecursiveIteratorIterator(
                new \RecursiveDirectoryIterator($dir, \RecursiveDirectoryIterator::SKIP_DOTS),
                \RecursiveIteratorIterator::CHILD_FIRST
            );
            foreach ($items as $item) {
                $item->isDir() ? @rmdir($item->getRealPath()) : @unlink($item->getRealPath());
            }
            @rmdir($dir);
        }

        // PERS-01: Slug generation
        public function testSlugFromQuestion(): void
        {

            $this->assertSame('what-are-the-latest-advances-in-transformer-architectures', $slug);
        }

        {
            $long = str_repeat('word ', 20);
            $slug = Manager::slugFromQuestion($long);
            $this->assertLessThanOrEqual(60, strlen($slug));
        }

    declare(strict_types=1);

    namespace App\Tests\Session;

    use App\Session\ProgressLogger;
    use PHPUnit\Framework\TestCase;

    class ProgressLoggerTest extends TestCase
    {
        private string $tempDir;

        protected function setUp(): void
        {
            $this->tempDir = sys_get_temp_dir() . '/research-agents-pl-test-' . bin2hex(random_bytes(4));
            @mkdir($this->tempDir, 0775, true);
        }

        protected function tearDown(): void
        {
            $logFile = $this->tempDir . '/session.log';
            if (file_exists($logFile)) { @unlink($logFile); }
            @rmdir($this->tempDir);
        }

        // LOG-03: Per-session log file created
        public function testLogEventCreatesFile(): void
        {
            $logFile = $this->tempDir . '/session.log';
            $logger = new ProgressLogger($logFile);
            $logger->logEvent('started', 'alpha', ['question' => 'test']);
            $this->assertFileExists($logFile);
        }

        public function testLogEventWritesValidJson(): void
        {
            $logFile = $this->tempDir . '/session.log';
            $logger = new ProgressLogger($logFile);
            $logger->logEvent('started', 'alpha', ['model' => 'gpt-4']);
            $line = trim(file_get_contents($logFile));
            $data = json_decode($line, true);
            $this->assertNotNull($data);
            $this->assertSame('alpha', $data['agent']);
            $this->assertSame('started', $data['event']);
            $this->assertArrayHasKey('ts', $data);
            $this->assertSame('PROGRESS', $data['channel']);
        }

        public function testReadNewLinesReturnsEvents(): void
        {
            $logFile = $this->tempDir . '/session.log';
            $logger = new ProgressLogger($logFile);
            $logger->logEvent('started', 'alpha');
            $logger->logEvent('web_search', 'beta');
            $result = $logger->readNewLines(0);
            $this->assertCount(2, $result['lines']);
            $this->assertGreaterThan(0, $result['new_offset']);
        }

        public function testReadNewLinesRespectsOffset(): void
        {
            $logFile = $this->tempDir . '/session.log';
            $logger = new ProgressLogger($logFile);
            $logger->logEvent('started', 'alpha');
            $logger->logEvent('web_search', 'beta');
            $firstResult = $logger->readNewLines(0);
            $secondResult = $logger->readNewLines($firstResult['new_offset']);
            $this->assertCount(1, $secondResult['lines']);
            $data = json_decode($secondResult['lines'][0], true);
            $this->assertSame('web_search', $data['event']);
        }

        public function testReadNewLinesFromMissingFile(): void
        {
            $logFile = $this->tempDir . '/nonexistent.log';
            $logger = new ProgressLogger($logFile);
            $result = $logger->readNewLines(0);
            $this->assertSame([], $result['lines']);
            $this->assertSame(0, $result['new_offset']);
        }

        public function testLogEventAllEventTypes(): void
        {
            $logFile = $this->tempDir . '/session.log';
            $logger = new ProgressLogger($logFile);
            $events = ['started', 'llm_call', 'web_search', 'paper_search', 'tool_result', 'score_evaluated', 'critique_r2_started', 'critique_completed', 'completed', 'timed_out', 'failed', 'batch_complete'];
            foreach ($events as $event) {
                $logger->logEvent($event, 'alpha');
            }
            $result = $logger->readNewLines(0);
            $this->assertCount(count($events), $result['lines']);
        }
    }

    Public method: createSession(string $question, array $data = []): string
    - Derives slug: $slug = self::slugFromQuestion($question)
    - Prefix with date: $sessionId = date('Y-m-d') . '_' . $slug
    - Creates directory: $this->sessionDir . '/' . $sessionId, mkdir(0775, true)
    - Generates transcript: calls $this->generateTranscript($question, $data['results'] ?? [], $data['debate'] ?? null, $data['duration_ms'] ?? 0)
    - Writes transcript to session.md: file_put_contents with LOCK_EX
    - Sets permissions: chmod(session.md, 0600) per T-05-01 (readable only by owner)
    - Returns $sessionId

    Public method: generateTranscript(string $question, array $results, ?array $debate, int $durationMs): string
    (Per D-08 and UI-SPEC.md Section 5.2)

    Builds string in this exact order:

    1. YAML frontmatter (between "---\n" delimiters):
       - question: (single line, newlines replaced with spaces)
       - date: (ISO 8601 date('c'))
       - agent_count: (count)
       - agents: [comma-separated names]
       - winner: (from $debate['winner'] ?? 'N/A')
       - correlation_id: (from first result's correlation_id)
       - duration_ms: (durationMs)
       - model_info: (space-separated unique model names from results)
       - score_summary: (for agents with scores in score_table, "name: weighted_total" pairs)

    2. ## Summary section:
       - **Question:** restated
       - **Winner:** bold winner name
       - **Duration:** in seconds
       - Score table as markdown table:
         | Agent | Quality | Critique | Diversity | Total |
         |-------|---------|----------|-----------|-------|
         For each agent in score_table: | name | quality/10 | critique_avg | diversity_bonus | weighted_total |
         (Skip score table if $debate is null or has no score_table)

    3. ## Raw Answers section:
       For each agent $name => $result in $results:
       ### {name}
       - **Model:** {model}
       - **Response time:** {response_time_ms}ms
       - **Tokens:** {usage.prompt_tokens} in / {usage.completion_tokens} out
       - **Error:** {error} (only if $result has 'error' key)
       Blank line, then $result['answer'], then blank line, then "---" separator.

    4. ## Debate section (only if $debate is not null):
       ### Winner: {debate.winner}
       {debate.narrative}

    5. ## Errors section (only if any result has 'error' key):
       For each result with error: - **{name}:** {error}

    Per D-05: "Full everything" -- every agent's complete raw answer, quality scores, structured critique, diversity bonuses, score table, winner answer, judge narrative, tool call results, and error summary.

    Public method: readSession(string $slug): ?array
    - Path: $this->sessionDir . '/' . basename($slug) . '/session.md'
    - Uses basename($slug) to prevent path traversal (T-05-03 -- defense in depth, also enforced in Plan 3/Plan 2)
    - If file does not exist, return null
    - Parse lines: find first "---", read frontmatter until next "---", parse as YAML-like key: value pairs
    - Split on ": " (first colon), trim values. Boolean: "true"/"false" -> bool. Numeric: cast to int/float. List: strip brackets, explode by ", "
    - Returns array with all frontmatter keys

    Security note: No session deletion or overwrite methods are implemented in v1. Per UI-SPEC.md Section 6: "No destructive actions exposed to user."

    Public method: logEvent(string $event, string $agent = '', array $data = []): void
    (Per UI-SPEC.md Section 5.3 -- Progress Log Format)

    Builds JSON object:
    ```php
    $entry = json_encode([
        'ts'      => (new \DateTimeImmutable())->format('Y-m-d\TH:i:s.v\Z'),
        'agent'   => $agent,
        'channel' => 'PROGRESS',
        'event'   => $event,
        'data'    => $data,
    ], JSON_UNESCAPED_SLASHES | JSON_UNESCAPED_UNICODE);
    ```

    Writes to file: file_put_contents($this->logFile, $entry . "\n", FILE_APPEND | LOCK_EX)

    FILE_APPEND | LOCK_EX is REQUIRED (Pitfall 4: concurrent writes from multiple children). LOCK_EX provides exclusive write access. Each line is under 1KB.

    Public method: readNewLines(int $offset): array
    (Per RESEARCH.md Code Examples -- tailFile function, lines 798-827)

    Returns: ['lines' => string[], 'new_offset' => int]

    If file does not exist: return ['lines' => [], 'new_offset' => 0]
    If filesize <= offset: return ['lines' => [], 'new_offset' => $size]
    Otherwise:
    - fopen($path, 'rb')
    - fseek($handle, $offset)
    - stream_get_contents($handle)
    - fclose($handle)
    - $lines = array_filter(explode("\n", $content), fn($l) => $l !== '')
    - return ['lines' => array_values($lines), 'new_offset' => $size]

    NOTE: readNewLines is explicitly NOT a blocking tail -f. It reads from a byte offset and returns all new content. The caller (CLI repl.php or SSE endpoint) is responsible for polling with sleep().
